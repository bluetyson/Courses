\paragraph{Problem 1}

Given a joint distribution, calculate various quantities:

\begin{tabular}[c]{l|c c}
\(y/x\) & 0 & 1 \\
\hline
0 & 1/4 & 1/4 \\
1 & 0 & 1/2 \\
\hline
\end{tabular}

\begin{enumerate}
\item H(X,Y)
\item H(X)
\item H(Y)
\item H(X\textbar{}Y)
\item H(Y\textbar{}X)
\item H(X) + H(Y) - H(X,Y)
\item H(X) - H(X\textbar{}Y)
\item H(Y) - H(Y\textbar{}X)
\item I(X;Y)
\end{enumerate}
